# input and store number
	IN
start	STO	input
	OUT
	SUB	one
	BRZ	halt

# reset count to zero
	LDA	zero
	STO	count
	LDA	input

# subtract two until either zero or negative
loop	SUB	two
	BRZ	even
	BRP	loop

# if odd number
	LDA	input
	ADD	input
	ADD	input
	ADD	one
	STO	input	
	BR	divide

# if even number
even	BR	divide

# divide by 2
divide	LDA	count
	ADD	one
	STO	count
	LDA	input
	SUB	two
	STO	input
	BRP	divide
# Subtract one from count
	LDA	count
	SUB	one
	BR	start

halt	HLT

# data initalisation
input	DAT	0
one	DAT	1
two	DAT	2
count	DAT	0
zero	DAT	0
