# input and store number
	IN
start	STO	f
	OUT
# check if one
	SUB	one
	BRZ	halt
# reset count to zero
	LDA	zero
	STO	count
# divide by two and check even/odd
subtwo	LDA	count
	ADD	one
	STO	count
	LDA	f
	SUB	two
	STO	f
# check if the final result was a zero (therefore even input)
	BRZ	even
# loop if not zero
	BRP	subtwo
# if odd number
	LDA	count
# check if over 334 (667 total)
	SUB	ttf
	BRP	over
# multiply by three and add two
	LDA	count
	ADD	count
	ADD	count
	SUB	one
	BR	start
# if even number
even	LDA	count
	BR	start
# overflow case
over	LDA	zero
	OUT
halt	HLT
# data initalisation
f	DAT	0
zero	DAT	0
one	DAT	1
two	DAT	2
ttf	DAT	334
count	DAT	0
