# input and store number
	IN
start	STO	input
	OUT
# check if one
	SUB	one
	BRZ	halt
# reset count to zero
	LDA	zero
	STO	count
# divide by two and check even/odd
divide	LDA	count
	ADD	one
	STO	count
	LDA	input
	SUB	two
	STO	input
	BRP	divide
# check if the final result was a zero (therefore even input)
	BRZ	even
# if odd number
	LDA	count
	SUB	one
	STO	input
# check if over 333 (667 total)
	SUB	ttt
	BRP	over
# multiply by three and add two
	LDA	input
	ADD	input
	ADD	input
	ADD	two
	BR	start
# if even number
even	LDA	count
	SUB	one
	BR	start
# overflow case
over	LDA	zero
	OUT
halt	HLT
# data initalisation
input	DAT	0
zero	DAT	0
one	DAT	1
two	DAT	2
ttt	DAT	333
count	DAT	0
